    <input type="text" id="method1">

    <input type="number" id="amount1">

   

    <label>Другий звіт:</label>

    <select id="report2">

        <option>Періодичний звіт</option>

        <option>Х звіт</option>

        <option>Z звіт</option>

        <option>Звіт на касі</option>

        <option>Статистика →</option>

    </select>

   

    <label>Сума (другий звіт):</label>

    <input type="number" id="amount2">

   

    <button onclick="generateText()">Сформувати текст</button>

    <button onclick="copyText()">Скопіювати текст</button>

   

    <div id="output"></div>

   

    <script>

        function generateText() {

            let date = document.getElementById('date').value;

            let place = document.getElementById('place').value;

            let report1 = document.getElementById('report1').value;

            let method1 = document.getElementById('method1').value;

            let amount1 = document.getElementById('amount1').value;

            let report2 = document.getElementById('report2').value;

            let amount2 = document.getElementById('amount2').value;

            let difference = Math.abs(amount1 - amount2);

           

            let text = `${date} в закладі ${place} сталась розбіжність:\n` +

                `В ${report1} сума по ${method1} — ${amount1} грн\n` +

                `Натомість, в ${report2} сума по ${method1} — ${amount2} грн\n` +

                `Розбіжність = ${difference} грн`;

           

            document.getElementById('output').innerText = text;

        }

       

        function copyText() {

            let text = document.getElementById('output').innerText;

            navigator.clipboard.writeText(text).then(() => {

                alert('Текст скопійовано!');

            });

        }

    </script>
